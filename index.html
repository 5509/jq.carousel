(function() {
  var $carousel = $('#carousel_2').carousel({
      start: 2
    }),
    $indicator = $('#indicator_2').find('span');

  $('#carousel_2_prev').on('click', function(ev) {
    ev.preventDefault();
    $carousel.carouselPrev();

    $indicator.removeClass('active');
    $indicator.eq($carousel.carouselGetCurrent()).addClass('active');
  });
  $('#carousel_2_next').on('click', function(ev) {
    ev.preventDefault();
    $carousel.carouselNext();

    $indicator.removeClass('active');
    $indicator.eq($carousel.carouselGetCurrent()).addClass('active');
  });
}());

source code

(function() {
  var $carousel = $('#carousel_4').carousel(),
    num = 1;

  $('#carousel_4_add').on('click', function(ev) {
    $carousel.carouselReset();
    $carousel.append('<div class="carousel_box">' + (++num) + '</div>');
    $carousel.carouselRefresh();
  });
  $('#carousel_4_prev').on('click', function(ev) {
    ev.preventDefault();
    $carousel.carouselPrev();
  });
  $('#carousel_4_next').on('click', function(ev) {
    ev.preventDefault();
    $carousel.carouselNext();
  });
}());
