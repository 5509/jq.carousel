reset and refresh carousel, and binding event

1

source code

(function() {
  var $carousel = $('#carousel_4').carousel({
      group: 4,
      indicator: true
    }),
    $prev = $('#carousel_4_prev'),
    $next = $('#carousel_4_next'),
    num = 1;

  $('#carousel_4_add').on('click', function(ev) {
    $carousel.carouselReset();
    $carousel.append('<div class="carousel_box">' + (++num) + '</div>');
    $carousel.carouselRefresh();
  });

  // ...

  $carousel.bind('Carousel.refresh', function() {
    if ( !$carousel.carouselGetMoveState() ) {
      return;
    }
    $prev.prop('disabled', '');
    $next.prop('disabled', '');
  });
}());

var $carousel = $('#carousel_6').carousel({
      loop: false,
      indicator: true
    }),
    $prev = $('#carousel_6_prev'),
    $next = $('#carousel_6_next'),
    set_state = function() {
      var state = $carousel.carouselGetMoveState();
      switch ( state ) {
      case 'min':
        $prev.prop('disabled', 'disabled');
        break;
      case 'max':
        $next.prop('disabled', 'disabled');
        break;
      case true:
        $prev.prop('disabled', '');
        $next.prop('disabled', '');
        break;
      default:
        $prev.prop('disabled', 'disabled');
        $next.prop('disabled', 'disabled');
      }
    };

  // ...

  $carousel.bind({
    'Carousel.next': function() {
      set_state();
    },
    'Carousel.prev': function() {
      set_state();
    }
  });

      group: 3,
      indicator: true
    }),
    $select = $('#carousel_8_select'),
    // ...

  // ...

  $select.on('change', function() {
    var val = this.value,
      group = val / 200;

    // if carousel width is changed?
    $carousel.carouselReset();
    $carousel.css('width', val);
    $carousel.carouselChangeConf({
      group: group
    });
    $carousel.carouselRefresh();
  });
}());
